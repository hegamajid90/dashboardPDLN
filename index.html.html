📋 Cara Setup Koneksi Google Sheets → Dashboard
1
Buka Apps Script
Di Google Sheets PDLN, klik Extensions → Apps Script. Hapus kode default, paste script dari kotak di bawah.
2
Deploy Web App
Klik Deploy → New deployment. Pilih type Web App. Execute as: Me. Access: Anyone.
3
Salin URL
Setelah deploy, salin Web App URL yang muncul (format: …/macros/s/…/exec).
4
Paste di Banner
Paste URL tersebut di kolom input pada banner merah di atas, lalu klik Hubungkan. Dashboard akan sync otomatis.
📄 Kode Apps Script — Copy & Paste ke Google Apps Script
function doGet(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet()
                  .getSheetByName("2026"); // Sesuaikan nama sheet tab
    var data  = sheet.getDataRange().getValues();
    var rows  = [];
    for (var i = 2; i < data.length; i++) {
      var r = data[i];
      if (!r[2]) continue; // skip baris kosong
      rows.push({
        nomorKemlu: r[2]||"",  nomorSetneg: r[1]||"",
        tanggal:    r[3]||"",  unit:        r[4]||"",
        negara:     r[5]||"",  awal:        r[6]||"",
        akhir:      r[7]||"",  delegasi:    r[8]||"",
        jenis:      r[9]||"",  kegiatan:    r[10]||"",
        simpel:     r[11]||"", status:      r[14]||"",
        laporan:    r[15]||""
      });
    }
    return ContentService
      .createTextOutput(JSON.stringify({status:200,data:rows,count:rows.length}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:500,error:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    var p     = JSON.parse(e.postData.contents);
    var sheet = SpreadsheetApp.getActiveSpreadsheet()
                  .getSheetByName("2026");
    sheet.appendRow(["", p.nomorSetneg||"", p.nomorKemlu||"",
      p.tanggal||"", p.unit||"", p.negara||"",
      p.awal||"", p.akhir||"", p.delegasi||"",
      p.jenis||"", p.kegiatan||"", p.simpel||"",
      "","", p.status||"Draft / Konsep", p.laporan||"Belum"]);
    return ContentService
      .createTextOutput(JSON.stringify({status:200,message:"OK"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:500,error:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
